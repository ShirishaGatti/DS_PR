int enqueue(int queue[10],int *front,int *rear,int data,int len)
 {
    if(*rear==len)
    {
        //full
        return -1;
    }
    else if(*front==-1&&*rear==-1)
    {
        //Empty  
        *front,*rear=0;
        queue[*rear]=data;
        return 1;
    }
    else
     {
         *rear++;
         queue[*rear]=data;
         return 1;
     }
 }
 int dequeue(int queue[10],int *front,int *rear,int len){

        return -1;
    }

    {

        val=queue[*front];
        *front++;
        return val;
    }
    else

 }
